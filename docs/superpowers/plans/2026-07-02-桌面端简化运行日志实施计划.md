# 桌面端简化运行日志 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 将桌面端任务日志面板从完整日志预览改为面向普通用户的简化运行流程视图，同时保留完整日志文件用于排查问题。

**Architecture:** 任务运行时继续写现有完整日志，并在同一个 `TaskTextLogger` 管线中并行写 `*.simple.txt` 简化日志，保证完整日志的每个流程段至少对应一条简化记录。前端仍通过 IPC 读取日志，但默认读取简化日志；另加一个 IPC 动作用系统方式打开完整日志文件。

**Tech Stack:** Electron main/preload IPC、React、TypeScript、Vitest、React Testing Library、Node `fs/promises`、Electron `shell`。

---

## 需求边界

- 完整日志保持现状，继续包含 prompt、工具参数、工具返回正文、错误详情。
- 用户面板不再展示完整日志正文，只展示简化日志。
- 简化日志必须实时刷新、自动下滑。
- 简化日志面向普通用户，但完整覆盖流程：完整日志每次 `append(tags, value)` 都要产生一条简化日志。
- `工具调用` 和 `工具返回` 必须逐条显示，各占一行，不合并同类工具。
- 面板保留 `打开完整日志` 入口，只打开或定位完整 `.txt` 文件，不在应用内预览完整日志。
- 不新增数据库表；简化日志作为完整日志的同目录派生产物，路径由完整日志路径推导。

## 架构与配置化审查

- **通用架构:** 新增独立的简化日志摘要模块 `taskProgressLog.ts`，不把摘要规则散落在 `windowRunService.ts` 或前端组件里。`TaskTextLogger` 负责双写，调用者仍只调用 `append(tags, value)`。
- **设计模式适配:** 使用“事件摘要器”模式：完整日志事件 `{ tags, value, timestamp }` 输入，输出一行用户可读文本。新增工具类型或流程标签时，只扩展摘要器映射。
- **配置化覆盖范围:** 当前摘要规则是产品文案与日志协议的一部分，暂不做外部配置；但常量集中在摘要模块中，避免硬编码散落。轮询间隔、自动滚动阈值作为组件内命名常量，后续可迁移到配置。
- **简单优先:** 不引入数据库、流式 IPC 或前端解析完整日志。并行写 `*.simple.txt` 最贴近现有文件日志架构。

```ts
import path from "node:path";

export interface TaskProgressLogEntry {
  tags: readonly string[];
  timestamp: string;
  value: unknown;
}

function asRecord(value: unknown): Record<string, unknown> | undefined {
  return value !== null && typeof value === "object" && !Array.isArray(value)
    ? (value as Record<string, unknown>)
    : undefined;
}

function shortTime(timestamp: string): string {
  const match = timestamp.match(/T?(\d{2}):(\d{2}):(\d{2})/u);
  return match ? `${match[1]}:${match[2]}:${match[3]}` : timestamp;
}

function fileName(value: unknown): string {
  if (typeof value !== "string" || value.trim() === "") {
    return "未指定文件";
  }
  return path.basename(value.replace(/\\/gu, "/"));
}

function countArray(value: unknown): number {
  return Array.isArray(value) ? value.length : 0;
}

function stringField(record: Record<string, unknown> | undefined, key: string): string | undefined {
  const value = record?.[key];
  return typeof value === "string" && value.trim() !== "" ? value : undefined;
}

function objectField(record: Record<string, unknown> | undefined, key: string): Record<string, unknown> | undefined {
  return asRecord(record?.[key]);
}

function toolPath(value: unknown): string {
  const record = asRecord(value);
  const args = objectField(record, "实际执行输入") ?? objectField(record, "模型原始输入") ?? record;
  const rawPath = args?.path ?? args?.pattern ?? args?.job_id;
  return fileName(rawPath);
}

function hasResultContent(value: unknown): boolean {
  if (typeof value === "string") {
    return value.trim() !== "" && value.trim() !== "[]";
  }
  if (Array.isArray(value)) {
    return value.length > 0;
  }
  return value !== null && typeof value !== "undefined";
}

function summarizeTaskInfo(value: unknown): string {
  const text = typeof value === "string" ? value : "";
  const book = /书籍\s*([^，]+)/u.exec(text)?.[1] ?? "未知书籍";
  const model = /模型\s*([^，]+)/u.exec(text)?.[1] ?? "未知模型";
  const templates = /模板\s*(\d+)\s*个/u.exec(text)?.[1];
  const templateText = templates ? `${templates} 个模板` : "模板未设置";
  return `开始任务：${book}，${templateText}，模型 ${model}`;
}

function summarizeToolCall(toolName: string, value: unknown): string {
  const target = toolPath(value);
  switch (toolName) {
    case "read_file":
      return `读取文件：${target}`;
    case "glob":
      return `查找报告文件：${target}`;
    case "grep":
      return `搜索文件：${target}`;
    case "ls":
      return `列出目录：${target}`;
    case "write_file":
      return `写入报告：${target}`;
    case "edit_file":
      return `更新报告：${target}`;
    case "multi_edit":
      return `批量更新报告：${target}`;
    case "mark_no_update":
      return `标记无新增：${target}`;
    case "bash":
      return "执行命令：bash";
    case "bash_output":
      return `读取后台命令输出：${target}`;
    case "wait":
      return "等待后台任务完成";
    case "kill_shell":
      return `终止后台任务：${target}`;
    default:
      return `调用工具 ${toolName}：${target}`;
  }
}

function summarizeToolResult(toolName: string, value: unknown): string {
  const record = asRecord(value);
  const target = toolPath(value);
  if (record?.["是否可恢复错误"] === true) {
    switch (toolName) {
      case "write_file":
        return `写入返回可恢复错误：${target}，模型将重试`;
      case "edit_file":
        return `更新返回可恢复错误：${target}，模型将重试`;
      case "multi_edit":
        return `批量更新返回可恢复错误：${target}，模型将重试`;
      case "bash":
        return "命令返回可恢复错误：bash，模型将重试";
      default:
        return `工具返回可恢复错误：${target}，模型将重试`;
    }
  }
  const hasContent = hasResultContent(record?.返回内容);
  switch (toolName) {
    case "read_file":
      return `读取完成：${target}`;
    case "glob":

    case "ls":

    case "edit_file":
      return `更新完成：${target}`;

function summarizeEntryBody(tags: readonly string[], value: unknown): string {
  const [primary, secondary] = tags;
  const record = asRecord(value);

  if (primary === "任务信息") {
    return summarizeTaskInfo(value);
  }
  if (primary === "上下文" && secondary === "任务") {
    return `加载任务上下文：${stringField(record, "书籍名称") ?? "书籍"}、规则快照和报告目录已准备`;
  }
  if (primary === "上下文" && secondary === "覆盖索引") {
    return `检查覆盖索引：跳过已提取=${record?.跳过已提取 === true ? "true" : "false"}，待处理模板 ${countArray(record?.待处理模板)} 个`;
  }
  if (primary === "上下文" && secondary === "窗口") {
    const windowText = stringField(record, "正在处理") ?? stringField(record, "窗口序号") ?? "窗口";
    const chapters = stringField(record, "章节范围") ?? "未知章节";
    return `${windowText}：处理第 ${chapters} 章，模板 ${countArray(record?.模板)} 个`;
  }
  if (primary === "大模型请求") {
    const windowText = stringField(record, "窗口") ?? "未知窗口";
    return `请求模型：窗口 ${windowText}，第 ${record?.轮次 ?? "?"} 轮`;
  }
  if (primary === "大模型返回") {
    const toolCount = countArray(record?.工具调用);
    return toolCount > 0 ? "模型返回：准备读取窗口文本和检查已有报告" : "模型返回：无工具调用";
  }
  if (primary === "工具调用" && secondary) {
    return summarizeToolCall(secondary, value);
  }
  if (primary === "工具返回" && secondary) {
    return summarizeToolResult(secondary, value);
  }
  if (primary === "上下文" && secondary === "重试") {
    const text = typeof value === "string" ? value : "";
    const missing = /缺少 outputFileName：(.+?)(?:。|$)/u.exec(text)?.[1];
    return missing ? `继续补齐结果：缺少 ${missing}` : "继续补齐结果：模型需要重试";
  }
  if (primary === "上下文" && secondary === "批次结果") {
    const outcomes = Array.isArray(record?.处理结果) ? record.处理结果 : [];
    const written = outcomes.filter((item) => asRecord(item)?.outcome !== "no_update").length;
    const noUpdate = outcomes.length - written;
    return `完成窗口 ${stringField(record, "窗口") ?? "当前窗口"}：写入 ${written} 个报告，标记 ${noUpdate} 个无新增`;
  }
  if (primary === "上下文" && secondary === "覆盖索引更新") {
    return `更新覆盖索引：窗口 ${stringField(record, "窗口") ?? "当前窗口"} 已记录`;
  }
  if (primary === "错误") {
    const reason = typeof value === "string" ? value : stringField(record, "原因") ?? "查看完整日志获取详情";
    if (secondary === "窗口") {
      return `窗口失败：窗口 ${stringField(record, "窗口") ?? "未知"}，原因 ${reason}`;
    }
    return `${secondary ?? "任务"}失败：原因 ${reason}`;
  }
  if (primary === "警告") {
    return `运行警告：${typeof value === "string" ? value : secondary ?? "查看完整日志获取详情"}`;
  }
  return `${tags.join("/")}：已记录`;
}

export function summarizeTaskLogEntry(entry: TaskProgressLogEntry): string {
  return `${shortTime(entry.timestamp)} ${summarizeEntryBody(entry.tags, entry.value)}`;
}
```

```ts
function deriveSimpleLogFilePath(logFilePath: string): string {
  return logFilePath.replace(/\.txt$/u, ".simple.txt");
}

async function readJobLog(job: P0JobRecord): Promise<string> {
  if (!job.logFilePath) {
    return "任务尚未开始，运行流程还没有生成。";
  }

  const book = requireBook(job.bookId);
  const project = await ensureProject(book.projectId);
  const simpleLogPath = path.join(project.rootPath, deriveSimpleLogFilePath(job.logFilePath));

  try {
    return await fs.readFile(simpleLogPath, "utf8");
  } catch (error) {
    if ((error as NodeJS.ErrnoException).code === "ENOENT") {
      return "运行流程文件不存在，可打开完整日志查看详情。";
    }
    throw error;
  }
}

async function openJobLog(job: P0JobRecord): Promise<void> {
  if (!job.logFilePath) {
    throw new Error("任务尚未开始，完整日志文件还没有生成。");
  }
  const book = requireBook(job.bookId);
  const project = await ensureProject(book.projectId);
  const logPath = path.join(project.rootPath, job.logFilePath);
  const result = await options.shell.openPath(logPath);
  if (result) {
    throw new Error(result);
  }
}
```

```tsx
export interface JobLogPanelProps {
  jobId: string;
  logFilePath?: string;
  onReadLog?: (jobId: string) => Promise<string>;
  onOpenLog?: (jobId: string) => Promise<void>;
}

export function JobLogPanel({ jobId, logFilePath, onReadLog, onOpenLog }: JobLogPanelProps) {
  const [isOpen, setOpen] = useState(false);
  const [content, setContent] = useState<string | undefined>();
  const [state, setState] = useState<"idle" | "loading" | "error">("idle");
  const [errorMessage, setErrorMessage] = useState<string | undefined>();
  const [isFollowing, setFollowing] = useState(true);
  const logRef = useRef<HTMLPreElement | null>(null);

  async function refreshLog(): Promise<void> {
    if (!onReadLog) {
      setContent("运行流程读取入口尚未就绪。");
      return;
    }
    setState((current) => (current === "idle" && !content ? "loading" : current));
    setErrorMessage(undefined);
    try {
      const nextContent = await onReadLog(jobId);
      setContent(nextContent || "运行流程暂无内容。");
      setState("idle");
    } catch (error) {
      setErrorMessage(error instanceof Error ? error.message : "运行流程读取失败，可打开完整日志查看详情。");
      setState("error");
    }
  }

  useEffect(() => {
    if (!isOpen) {
      return undefined;
    }
    void refreshLog();
    const timer = window.setInterval(() => {
      void refreshLog();
    }, LOG_REFRESH_MS);
    return () => window.clearInterval(timer);
  }, [isOpen, jobId, onReadLog]);

  useEffect(() => {
    const element = logRef.current;
    if (element && isFollowing) {
      element.scrollTop = element.scrollHeight;
    }
  }, [content, isFollowing]);

  function handleScroll(): void {
    const element = logRef.current;
    if (!element) {
      return;
    }
    const distanceToBottom = element.scrollHeight - element.scrollTop - element.clientHeight;
    setFollowing(distanceToBottom <= AUTO_SCROLL_EPSILON);
  }

  if (!logFilePath) {
    return <p className="job-log-summary">暂无运行流程</p>;
  }

  return (
    <div className="job-log-panel">
      <div className="job-log-actions">
        <button className="button button--quiet" onClick={() => setOpen((current) => !current)} type="button">
          {isOpen ? "收起流程" : "展开流程"}
        </button>
        <button
          className="button button--quiet"
          onClick={() => {
            void onOpenLog?.(jobId);
          }}
          type="button"
        >
          打开完整日志
        </button>
      </div>
      {isOpen ? (
        <div className="job-log-content">
          {state === "loading" ? <p className="job-log-summary">正在读取运行流程</p> : null}
          {state === "error" ? <p className="job-log-error">{errorMessage}</p> : null}
          {content ? (
            <pre className="job-log-text" onScroll={handleScroll} ref={logRef}>
              {content}
            </pre>
          ) : null}
        </div>
      ) : null}
    </div>
  );
}
```
