# 小说提取界面重构 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 按 `docs/superpowers/specs/2026-07-02-小说提取界面重构设计.md` 重构桌面端小说提取界面：实现自定义窗口外壳、左侧图标导航、保留底部状态栏空壳、支持 `.txt/.md` 上传、紧凑文件输入区、原位置模板上传、分组参数面板、带筛选和高级信息的任务队列卡片。

**Architecture:** UI 分三层推进。第一层是桌面外壳层：Electron 无边框窗口、窗口控制 IPC、标题栏、左侧导航、底部状态栏。第二层是业务数据层：上传源格式、任务 DTO、运行时任务记录、进度和耗时结构化字段。第三层是页面表现层：提取页三栏布局、上传面板、参数面板、任务队列 view model 和卡片样式。所有任务状态、筛选项、图标配置、动作按钮配置放在 renderer 配置或 view model 中，避免在 JSX 中散落硬编码。

**Tech Stack:** Electron main/preload、React、TypeScript、Vitest、Testing Library、CSS、`lucide-react`。

## 最终验收标准

- [ ] 顶部窗口栏是应用自绘标题栏，显示 `NovelExtractor` 和窗口控制按钮；页面标题下面不再显示“今日任务 / 进行中 / 已完成 / 失败”四个统计卡。
- [ ] 左侧导航只显示图标；鼠标悬停或键盘聚焦显示名称 tooltip；悬停时图标轻微放大，不造成布局位移；`prefers-reduced-motion` 下关闭动画。
- [ ] 底部状态栏保留高度和边框，但当前不显示模型、并发、版本、API 连接状态等内容。
- [ ] 小说上传支持 `.txt` 和 `.md`；`.md` 按纯文本处理，内部仍保存为 `original.txt`，界面显示用户上传的原始文件名。
- [ ] 上传区域高度明显收敛，目标高度约 `120px`，包含上传图标、格式提示和选择按钮。
- [ ] 模板上传入口仍在旧 UI 的文件输入面板下方位置，不改成参考图中的单一模板下拉块。
- [ ] 任务筛选顺序是 `全部`、`进行中`、`暂停`、`失败`、`已完成`；不显示“清空已完成”按钮。
- [ ] 任务卡展示结构化高级信息：百分比、已用时、预计剩余、完成时间、打开日志、打开输出目录、暂停/恢复/重新运行等已有动作。
- [ ] “打开输出目录”打开最终报告目录，不打开运行临时目录。
- [ ] 删除任务只删除任务记录，不删除报告文件和报告资产；同一本书的报告仍可读取和预览。
- [ ] 进行中任务按“已完成窗口平均耗时 × 剩余窗口数”估算剩余时间；样本不足时显示“计算中”；暂停任务冻结最后一次估算并标注“已暂停”。
- [ ] 现有提取、模板上传、日志、任务动作流程不回退。
- [ ] 现有功能全部保留；如果参考图、布局重构或新交互与当前已有功能冲突，优先适配并保留当前功能，本次重构只改变显示方式、布局组织和必要的数据展示，不删除或降级已有能力。

```ts
function clampPercent(completedWindowCount: number, totalWindowCount: number): number {
  if (totalWindowCount <= 0) {
    return 0;
  }

  return Math.max(0, Math.min(100, Math.round((completedWindowCount / totalWindowCount) * 100)));
}

function toJobProgressDto(job: P0JobRecord): JobProgressDto | undefined {
  if (!job.progress) {
    return undefined;
  }

  return {
    completedWindowCount: job.progress.completedWindowCount,
    totalWindowCount: job.progress.totalWindowCount,
    percent: clampPercent(job.progress.completedWindowCount, job.progress.totalWindowCount)
  };
}

function toJobTimingDto(job: P0JobRecord, nowIso: string): JobTimingDto | undefined {
  if (!job.timing?.startedAt) {
    return undefined;
  }

  const startedAtMs = Date.parse(job.timing.startedAt);
  const completedAtMs = job.timing.completedAt ? Date.parse(job.timing.completedAt) : undefined;
  const nowMs = Date.parse(nowIso);
  const endMs = completedAtMs ?? nowMs;
  const elapsedMs = Number.isFinite(startedAtMs) && Number.isFinite(endMs) ? Math.max(0, endMs - startedAtMs) : undefined;

  if (job.status === "paused" && job.timing.estimatedRemainingMs !== undefined) {
    return {
      startedAt: job.timing.startedAt,
      completedAt: job.timing.completedAt,
      elapsedMs,
      estimatedRemainingMs: job.timing.estimatedRemainingMs,
      estimateState: "frozen"
    };
  }

  if (job.timing.estimatedRemainingMs !== undefined) {
    return {
      startedAt: job.timing.startedAt,
      completedAt: job.timing.completedAt,
      elapsedMs,
      estimatedRemainingMs: job.timing.estimatedRemainingMs,
      estimateState: "available"
    };
  }

  return {
    startedAt: job.timing.startedAt,
    completedAt: job.timing.completedAt,
    elapsedMs,
    estimateState: job.status === "running" ? "calculating" : "unknown"
  };
}

function toInputSummaryDto(
  job: P0JobRecord,
  book: Book | undefined,
  templates: TemplateDto[]
): InputSummaryDto | undefined {
  if (!book) {
    return undefined;
  }

  const templateNames = job.input.templateIds
    .map((id) => templates.find((t) => t.id === id)?.name)
    .filter((name): name is string => name !== undefined);

  return {
    bookDisplayName: book.displayName,
    templateNames,
    modelId: job.input.modelId
  };
}

async function buildJobDto(job: P0JobRecord): Promise<JobDto> {
  const nowIso = clock.now();
  const book = requireBook(job.bookId);
  const templates = await resolveJobTemplates(job);

  return {
    id: job.id,
    bookId: job.bookId,
    status: job.status,
    progressText: job.progressText,
    progress: toJobProgressDto(job),
    timing: toJobTimingDto(job, nowIso),
    output: {
      outputDirectoryLabel: book.displayName,
      canOpenOutputDirectory: job.status === "completed"
    },
    inputSummary: toInputSummaryDto(job, book, templates),
    tokenText: job.tokenText,
    failureReason: job.failureReason,
    logFilePath: job.logFilePath,
    "allowedActions": getAllowedActions(job.status),
    createdAt: job.createdAt,
    "updatedAt": job.updatedAt
  };
}

function toJobPatchFromRuntimeState(
  state: JobRuntimeState,
  previousJob: P0JobRecord | undefined,
  clock: Clock
): Partial<P0JobRecord> {
  const nowIso = clock.now();
  const patch: Partial<P0JobRecord> = {
    status: toJobStatusFromRuntime(state.status),
    progressText: formatRuntimeProgress(state),
    tokenText: formatTokenText(state.usage),
    failureReason: state.failureReason,
    progress: {
      completedWindowCount: state.completedWindowCount,
      totalWindowCount: state.totalWindowCount
    }
  };

  // 初始化或更新 timing
  const previousTiming = previousJob?.timing;
  patch.timing = { ...previousTiming };

  // 首次 running 时写入 startedAt
  if (state.status === "running" && !previousTiming?.startedAt) {
    patch.timing.startedAt = nowIso;
  }

  // completed/failed 时写入 completedAt
  if ((state.status === "completed" || state.status === "failed") && !previousTiming?.completedAt) {
    patch.timing.completedAt = nowIso;
  }

  // 计算预估剩余时间
  if (
    state.status === "running" &&
    state.completedWindowCount > 0 &&
    state.completedWindowCount < state.totalWindowCount &&
    patch.timing.startedAt
  ) {
    const startedAtMs = Date.parse(patch.timing.startedAt);
    const nowMs = Date.parse(nowIso);
    const elapsedMs = nowMs - startedAtMs;
    const avgMsPerWindow = elapsedMs / state.completedWindowCount;
    patch.timing.estimatedRemainingMs = Math.floor(
      avgMsPerWindow * (state.totalWindowCount - state.completedWindowCount)
    );
  }

  // paused 时冻结估算
  if (state.status === "paused" && previousTiming?.estimatedRemainingMs !== undefined) {
    patch.timing.estimateFrozenAt = nowIso;
  }

  return patch;
}
```

  { key: "paused", label: "暂停" },

export function getFilterCount(jobs: readonly ExtractionJob[], filter: JobQueueFilter): number {
  if (filter === "all") {
    return jobs.length;
  }

  return jobs.filter((job) => job.status === filter).length;
}

export function filterJobs(jobs: readonly ExtractionJob[], filter: JobQueueFilter): ExtractionJob[] {
  if (filter === "all") {
    return [...jobs];
  }

  return jobs.filter((job) => job.status === filter);
}

export function formatDuration(milliseconds?: number): string {
  if (milliseconds === undefined) {
    return "--";
  }

  const totalSeconds = Math.max(0, Math.floor(milliseconds / 1000));
  const hours = Math.floor(totalSeconds / 3600);
  const minutes = Math.floor((totalSeconds % 3600) / 60);
  const seconds = totalSeconds % 60;

  if (hours > 0) {
    return `${hours.toString().padStart(2, "0")}:${minutes.toString().padStart(2, "0")}:${seconds.toString().padStart(2, "0")}`;
  }

  return `${minutes.toString().padStart(2, "0")}:${seconds.toString().padStart(2, "0")}`;
}

export function getRemainingTimeLabel(job: ExtractionJob): string {
  if (!job.timing) {
    return "--";
  }

  if (job.timing.estimateState === "frozen") {
    return `已暂停 ${formatDuration(job.timing.estimatedRemainingMs)}`;
  }

  if (job.timing.estimateState === "calculating") {
    return "计算中";
  }

  if (job.timing.estimatedRemainingMs !== undefined) {
    return formatDuration(job.timing.estimatedRemainingMs);
  }

  return "--";
}
