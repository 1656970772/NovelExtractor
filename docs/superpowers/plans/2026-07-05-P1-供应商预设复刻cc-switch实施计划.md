在 `packages/config/src/configInvariants.ts` 中新增 provider 专属 helper，替换 `assertValidConfigInvariants()` 开头现有的 provider id/model id 校验块，避免继续把 provider 规则散落在主函数里：

```ts
const ALLOWED_PROVIDER_API_FORMATS = new Set(["openai_chat", "openai_responses"]);
const ALLOWED_REASONING_EFFORT_PARAMS = new Set(["reasoning_effort", "none"]);
const ALLOWED_REASONING_OUTPUT_FORMATS = new Set(["reasoning_content"]);

function assertProviderReasoning(value: unknown, label: string): void {
  assertConfigObject(value, label);
  const reasoning = value as Record<string, unknown>;

  assertBoolean(reasoning.supportsThinking, `${label} supportsThinking`);
  assertBoolean(reasoning.supportsEffort, `${label} supportsEffort`);
  assertNonEmpty(reasoning.thinkingParam, `${label} thinkingParam`);
  assertAllowedValue(String(reasoning.thinkingParam), new Set(["thinking"]), `${label} thinkingParam`);
  assertNonEmpty(reasoning.effortParam, `${label} effortParam`);
  assertAllowedValue(
    String(reasoning.effortParam),
    ALLOWED_REASONING_EFFORT_PARAMS,
    `${label} effortParam`
  );
  assertNonEmpty(reasoning.outputFormat, `${label} outputFormat`);
  assertAllowedValue(
    String(reasoning.outputFormat),
    ALLOWED_REASONING_OUTPUT_FORMATS,
    `${label} outputFormat`
  );

  if (reasoning.effortValueMode !== undefined) {
    assertAllowedValue(String(reasoning.effortValueMode), new Set(["deepseek"]), `${label} effortValueMode`);
  }
}

function assertProviderPresets(config: NovelExtractorConfig): void {
  assertUnique(
    config.providerPresets.map((provider) => provider.id),
    "provider preset id"
  );

  for (const provider of config.providerPresets) {
    assertAllowedValue(provider.apiFormat, ALLOWED_PROVIDER_API_FORMATS, "provider api format");

    if (provider.reasoning !== undefined) {
      assertProviderReasoning(provider.reasoning, `provider reasoning for ${provider.id}`);
    }

    if (!provider.allowsUserModels) {
      if (!provider.baseUrl?.trim()) {
        throw new ConfigInvariantError(`provider ${provider.id} requires a base URL`);
      }

      if (!Array.isArray(provider.endpointCandidates) || !provider.endpointCandidates.includes(provider.baseUrl)) {
        throw new ConfigInvariantError(`provider ${provider.id} endpoint candidates must include base URL`);
      }
    }

    assertUnique(
      provider.models.map((model) => model.id),
      `model id for provider ${provider.id}`
    );
    for (const model of provider.models) {
      assertNonEmpty(model.id, `model id for provider ${provider.id}`);
    }
  }
}
```

```ts
import { redactSecrets } from "./credentials";

export interface FetchedProviderModel {
  id: string;
  ownedBy?: string;
}

export interface BuildModelListUrlCandidatesInput {
  baseUrl: string;
  isFullUrl?: boolean;
  modelsUrl?: string;
}

export interface FetchModelsFromProviderInput extends BuildModelListUrlCandidatesInput {
  apiKey: string;
  userAgent?: string;
  fetch?: typeof fetch;
}

const ERROR_BODY_MAX_CHARS = 512;
const FETCH_TIMEOUT_MS = 15_000;

const KNOWN_COMPAT_SUFFIXES = [
  "/api/claudecode",
  "/api/anthropic",
  "/apps/anthropic",
  "/api/coding",
  "/claudecode",
  "/anthropic",
  "/step_plan",
  "/coding",
  "/claude"
];

function truncateBody(body: string): string {
  const chars = Array.from(body);
  return chars.length <= ERROR_BODY_MAX_CHARS
    ? body
    : `${chars.slice(0, ERROR_BODY_MAX_CHARS).join("")}…`;
}

function endsWithVersionSegment(url: string): boolean {
  const last = url.split("/").at(-1) ?? "";
  return /^v\d+$/u.test(last);
}

function stripCompatSuffix(baseUrl: string): string | undefined {
  for (const suffix of KNOWN_COMPAT_SUFFIXES) {
    if (baseUrl.endsWith(suffix)) {
      return baseUrl.slice(0, -suffix.length);
    }
  }

  return undefined;
}

function pushUnique(values: string[], value: string): void {
  if (!values.includes(value)) {
    values.push(value);
  }
}

export function buildModelListUrlCandidates(
  input: BuildModelListUrlCandidatesInput
): string[] {
  const override = input.modelsUrl?.trim();
  if (override) {
    return [override];
  }

  const trimmed = input.baseUrl.trim().replace(/\/+$/u, "");
  if (!trimmed) {
    throw new Error("Base URL is empty");
  }

  const candidates: string[] = [];

  if (input.isFullUrl) {
    const v1Index = trimmed.indexOf("/v1/");
    if (v1Index >= 0) {
      pushUnique(candidates, `${trimmed.slice(0, v1Index)}/v1/models`);
    } else {
      const slashIndex = trimmed.lastIndexOf("/");
      const schemeIndex = trimmed.indexOf("://");
      const root = slashIndex > schemeIndex + 2 ? trimmed.slice(0, slashIndex) : "";
      if (root) {
        pushUnique(candidates, `${root}/v1/models`);
      }
    }

    if (candidates.length === 0) {
      throw new Error("Cannot derive models endpoint from full URL");
    }

    return candidates;
  }

  if (endsWithVersionSegment(trimmed)) {
    pushUnique(candidates, `${trimmed}/models`);
    if (!trimmed.endsWith("/v1")) {
      pushUnique(candidates, `${trimmed}/v1/models`);
    }
  } else {
    pushUnique(candidates, `${trimmed}/v1/models`);
  }

  const stripped = stripCompatSuffix(trimmed)?.replace(/\/+$/u, "");
  if (stripped && stripped.includes("://")) {
    pushUnique(candidates, `${stripped}/v1/models`);
    pushUnique(candidates, `${stripped}/models`);
  }

  return candidates;
}

function asRecord(value: unknown): Record<string, unknown> | undefined {
  return value !== null && typeof value === "object" && !Array.isArray(value)
    ? value as Record<string, unknown>
    : undefined;
}

export function parseFetchedModels(value: unknown): FetchedProviderModel[] {
  const data = asRecord(value)?.data;
  if (!Array.isArray(data)) {
    return [];
  }

  return data
    .flatMap((entry) => {
      const record = asRecord(entry);
      const id = record?.id;
      if (typeof id !== "string" || !id.trim()) {
        return [];
      }

      const ownedBy = record?.owned_by;
      return [{
        id: id.trim(),
        ...(typeof ownedBy === "string" && ownedBy.trim() ? { ownedBy: ownedBy.trim() } : {})
      }];
    })
    .sort((left, right) => left.id.localeCompare(right.id));
}

async function readResponseBody(response: Response): Promise<unknown> {
  const text = await response.text();
  if (!text) {
    return undefined;
  }

  try {
    return JSON.parse(text);
  } catch {
    return text;
  }
}

async function fetchWithTimeout(
  fetcher: typeof fetch,
  url: string,
  init: RequestInit
): Promise<Response> {
  const controller = new AbortController();
  const timeout = setTimeout(() => controller.abort(), FETCH_TIMEOUT_MS);

  try {
    return await fetcher(url, { ...init, signal: controller.signal });
  } finally {
    clearTimeout(timeout);
  }
}

export async function fetchModelsFromProvider(
  input: FetchModelsFromProviderInput
): Promise<FetchedProviderModel[]> {
  const apiKey = input.apiKey.trim();
  if (!apiKey) {
    throw new Error("API Key is required to fetch models");
  }

  const fetcher = input.fetch ?? globalThis.fetch;
  if (!fetcher) {
    throw new Error("fetch is not available");
  }

  const candidates = buildModelListUrlCandidates(input);
  let lastCandidateError = "no candidates";

  for (const url of candidates) {
    const headers: Record<string, string> = {
      Authorization: `Bearer ${apiKey}`
    };
    const userAgent = input.userAgent?.trim();
    if (userAgent) {
      headers["User-Agent"] = userAgent;
    }

    const response = await fetchWithTimeout(fetcher, url, {
      method: "GET",
      headers
    });

    const body = await readResponseBody(response);
    if (response.ok) {
      return parseFetchedModels(body);
    }

    const bodyText = typeof body === "string" ? body : JSON.stringify(body);
    const safeBody = redactSecrets(truncateBody(bodyText ?? ""), [apiKey]);
    const message = `HTTP ${response.status}: ${safeBody}`;

    if (response.status === 404 || response.status === 405) {
      lastCandidateError = message;
      continue;
    }

    throw new Error(message);
  }

  throw new Error(`All candidates failed: ${lastCandidateError}`);
}
```

```ts
export function mergeFetchedModelsIntoForm(
  state: ProviderFormState,
  fetchedModels: readonly FetchedProviderModelDto[]
): ProviderFormState {
  const existingById = new Map(state.models.map((model) => [model.id, model]));

  for (const fetchedModel of fetchedModels) {
    const existing = existingById.get(fetchedModel.id);
    existingById.set(fetchedModel.id, existing ?? {
      id: fetchedModel.id,
      displayName: fetchedModel.id,
      enabled: true,
      isDefault: false
    });
  }

  const models = [...existingById.values()];
  const hasDefault = models.some((model) => model.id === state.modelName);
  const nextDefault = hasDefault ? state.modelName : models[0]?.id ?? state.modelName;

  return {
    ...state,
    modelName: nextDefault,
    models: models.map((model) => ({
      ...model,
      isDefault: model.id === nextDefault
    })),
    modelFetchState: "idle",
    modelFetchError: undefined
  };
}
```

```ts
function responsesUrl(baseUrl: string): string {
  return `${baseUrl.replace(/\/+$/, "")}/responses`;
}

interface OpenAiResponsesBody {
  output_text?: unknown;
  output?: unknown;
  usage?: unknown;
}

interface OpenAiResponsesTool {
  type: "function";
  name: string;
  description?: string;
  parameters?: Record<string, unknown>;
}

type OpenAiResponsesInputItem =
  | ChatCompletionContentMessage
  | { role: "assistant"; content: string }
  | { type: "function_call"; call_id: string; name: string; arguments: string }
  | { type: "function_call_output"; call_id: string; output: string };

function toResponsesTool(tool: ToolSchema): OpenAiResponsesTool {
  return {
    type: "function",
    name: tool.function.name,
    ...(tool.function.description ? { description: tool.function.description } : {}),
    ...(tool.function.parameters ? { parameters: tool.function.parameters } : {})
  };
}

function serializeResponsesInput(messages: readonly ChatCompletionMessage[]): OpenAiResponsesInputItem[] {
  return messages.flatMap((message) => {
    if (message.role === "assistant") {
      const items: OpenAiResponsesInputItem[] = message.content
        ? [{ role: "assistant", content: message.content }]
        : [];
      return [
        ...items,
        ...(message.toolCalls ?? []).map((toolCall) => ({
          type: "function_call" as const,
          call_id: toolCall.id,
          name: toolCall.name,
          arguments: serializeToolCallArguments(toolCall.arguments)
        }))
      ];
    }

    if (message.role === "tool") {
      return [{
        type: "function_call_output",
        call_id: message.toolCallId,
        output: message.content
      }];
    }

    return [message];
  });
}

function extractResponsesText(body: OpenAiResponsesBody): string {
  if (typeof body.output_text === "string") {
    return body.output_text;
  }

  const output = Array.isArray(body.output) ? body.output : [];
  return output
    .flatMap((item) => {
      const record = asRecord(item);
      const content = Array.isArray(record?.content) ? record.content : [];
      return content.flatMap((contentItem) => {
        const contentRecord = asRecord(contentItem);
        return typeof contentRecord?.text === "string" ? [contentRecord.text] : [];
      });
    })
    .join("");
}

function parseResponsesToolCalls(output: unknown): ToolCall[] {
  if (!Array.isArray(output)) {
    return [];
  }

  return output.flatMap((item) => {
    const record = asRecord(item);
    if (record?.type !== "function_call") {
      return [];
    }

    const name = record.name;
    if (typeof name !== "string" || !name.trim()) {
      return [];
    }

    return [{
      id: typeof record.call_id === "string" ? record.call_id : "",
      name,
      arguments: parseToolCallArguments(record.arguments)
    }];
  });
}
```
